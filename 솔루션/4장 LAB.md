```python
def print_root(a, b, c):
  d = (b ** 2) - (4 * a * c)
  root_big = (-b + d ** 0.5) / (2 * a)
  root_small = (-b - d ** 0.5) / (2 * a)
  print('해는', root_big, '또는', root_small)

print_root(1, 4, -21)
# 출력: 해는 3.0 또는 -7.0
```

```python
def circle_area_circum(radius):
    area = 3.14 * radius ** 2
    circum = 2 * 3.14 * radius
    return area, circum

radius = 10
area, circum = circle_area_circum(radius)

print('반지름', radius, '인 원의 면적은', area, '원의 둘레는', circum)
```

```python
def multiplies(n, m):
    result = []
    for i in range(1, m+1):
        result = result + [n * i]
    return result

r1, r2, r3, r4 = multiplies(3, 4)
print(r1, r2, r3, r4)
```

```python
def sum_nums(*args):
    print(len(args), '개의 인자', args)
    s = 0
    for arg in args:
        s = s + arg
    avg = s / len(args)
    print('합계 :', s, '평균 :', avg)

sum_nums(10, 20, 30)
# 출력: 3 개의 인자 (10, 20, 30)

sum_nums(10, 20, 30, 40, 50)
# 출력: 5 개의 인자 (10, 20, 30, 40, 50)

def min_nums(*args):
    m = args[0]
    for arg in args:
        if m > arg:
            m = arg
    print('최소값은', m)
    
min_nums(20, 40, 50, 10)
# 출력: 최소값은 10
```
